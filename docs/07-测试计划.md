### 6.3 测试脚本示例

```javascript
// k6 报工接口压测脚本示例
import http from 'k6/http';
import { check, sleep } from 'k6';

export const options = {
  stages: [
    { duration: '1m', target: 20 },   // 热身
    { duration: '3m', target: 100 },  // 爬升
    { duration: '3m', target: 100 },  // 峰值保持
    { duration: '1m', target: 0 },    // 降冷
  ],
  thresholds: {
    http_req_duration: ['p(95)<500'],
    http_req_failed: ['rate<0.001'],
  },
};

export default function () {
  const payload = JSON.stringify({
    work_order_id: 300001,
    step_id: 400001,
    workstation_id: 500001,
    operator_id: 'USER001',
    report_type: 'COMPLETE',
    good_qty: 10,
    scrap_qty: 0,
    rework_qty: 0,
    duration_min: 5,
  });

  const res = http.post('http://localhost:5180/api/v1/work-reports', payload, {
    headers: { 'Content-Type': 'application/json' },
  });

  check(res, { 'status is 200': (r) => r.status === 200 });
  sleep(0.5);
}
```

## 7. 验收测试（端到端）

### 7.1 核心流程冒烟测试

```
场景：完整的生产批次

1. 登录系统
2. 创建物料 A（成品）和物料 B、C（子件）
3. 维护 BOM：A = B×2 + C×1
4. 维护工艺路线：A 的工艺 = 工序10(上料) → 工序20(组装) → 工序30(测试)
5. 创建工单：生产 A × 100
6. 下达工单 → 系统生成三个工序任务
7. 排产到产线 1
8. 工序10 报工：good=100 → 工序10 完成
9. 工序20 报工：good=95, scrap=3, rework=2 → 工序20 完成
10. 工序30 质检：process_qc=90, scrap=5 → 首次质检不合格
11. 返工 5 件 → 再次报好工
12. 关闭工单 → 工单 COMPLETED
13. 按批次号追溯：原料 B、C 的去向
14. 按序列号追溯：成品 A 的加工记录
```
